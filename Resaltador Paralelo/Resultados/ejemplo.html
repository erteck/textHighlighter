def createPatterns(line):
    result = ""
    concat = [""]
    lastSymbol = [""]
    for i in range(0, len(line)):
        if line[i] == ')' and line[i - 1] != "\\":
            concat.pop()
            result += lastSymbol[-1]
            lastSymbol.pop()
        elif line[i:i + 2] == "(.":
            concat.append('')
            lastSymbol.append('')
        elif line[i:i + 2] == "(*":
            concat.append('')
            lastSymbol.append('*')
        elif line[i:i + 2] == "(+":
            concat.append('|')
            lastSymbol.append(')')
            result+= '('
        elif line[i:i + 2] == "(#":
            concat.append('')
            lastSymbol.append('+')
        elif line[i] == " " and line[i - 2:i] in ["(*", '(.', '(#', '(+']:
            result += "" + ''
        elif line[i] == " ":
            if line[i+1] !="8":
                result += concat[-1] + ""
        elif line[i - 1:i + 1] not in ["(*", '(.', '(#', '(+']:
            #print(concat[-1])
            if(line[i] == '$'):
                result = result + '.*'
            elif (line[i] == '8'):
                lastSymbol.pop()
                lastSymbol.append(')?')
            else:
                result = result + line[i]

    return result
